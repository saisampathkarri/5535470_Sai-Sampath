#include <stdio.h>
int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int m, n, i, j;
        scanf("%d", &m);
        scanf("%d", &n);
        int cost[n];
        for (i = 0; i < n; i++) scanf("%d", &cost[i]);
        for (i = 0; i < n; i++) {
            for (j = i + 1; j < n; j++) {
                if (cost[i] + cost[j] == m) {
                    printf("%d %d\n", i + 1, j + 1);
                    i = n;
                    break;
                }
            }
        }
    }
    return 0;
}
