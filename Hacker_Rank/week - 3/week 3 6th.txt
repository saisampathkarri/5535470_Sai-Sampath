#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int ranked[n];
    for (int i = 0; i < n; i++) scanf("%d", &ranked[i]);
    int m;
    scanf("%d", &m);
    int player[m];
    for (int i = 0; i < m; i++) scanf("%d", &player[i]);

    int distinct[n], dcount = 0;
    distinct[dcount++] = ranked[0];
    for (int i = 1; i < n; i++)
        if (ranked[i] != ranked[i - 1]) distinct[dcount++] = ranked[i];

    int idx = dcount - 1;
    for (int i = 0; i < m; i++) {
        while (idx >= 0 && player[i] >= distinct[idx]) idx--;
        printf("%d\n", idx + 2);
    }
    return 0;
}
