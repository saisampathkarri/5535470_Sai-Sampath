#include <stdio.h>
#include <string.h>
#define MAXC 26
const char* isValid(char* s) {
    int freq[MAXC] = {0};
    for (int i = 0; s[i]; i++) freq[s[i] - 'a']++;

    int f1 = 0, f2 = 0, c1 = 0, c2 = 0;
    for (int i = 0; i < MAXC; i++) {
        if (freq[i] == 0) continue;
        if (f1 == 0 || freq[i] == f1) {
            f1 = freq[i];
            c1++;
        } else if (f2 == 0 || freq[i] == f2) {
            f2 = freq[i];
            c2++;
        } else {
            return "NO";
        }
    }

    if (f2 == 0) return "YES"; 
    if ((c1 == 1 && (f1 - 1 == f2 || f1 == 1)) ||
        (c2 == 1 && (f2 - 1 == f1 || f2 == 1)))
        return "YES";

    return "NO";
}

int main() {
    char s[100005];
    scanf("%s", s);
    printf("%s\n", isValid(s));
    return 0;
}
